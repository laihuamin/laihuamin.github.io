##let和const命令
一、基本用法（略）
二、无变量提升

console.log(foo); //undefined
console.log(tag); //Uncaught ReferenceError
var foo;
let tag;
三、暂时性死区(TDZ)

var foo = 13;
if (true) {
foo = ‘abc’; //ReferenceError
let foo;
}
四、不能重复声明

//报错
function () {
var foo;
let foo;
}
//报错
function (arg) {
let arg
}
//不报错
function (arg) {
{
let arg;
}
}

##解构
用途：
1、交换变量

[x, y] = [y, x];
2、返回多个参数

function example() {
return [1, 2, 3]
}
var [a, b, c] = example();
3、定义多个参数

function f([x, y, z]) {}
f([1, 2, 3]);// 有序
function g({x, y, z}) {}
g({z: 3, x: 1, y: 2}); // 无序
4、提取json数据

var jsonData = {
id: 24,
k: 546,
data: [123,456]
};
var{id, k, data: number} = jsonData;
5、函数默认参数值

jQuery.ajax = function (url, {
async = true,
beforeSend = function () {},
cache = true,
complete = function () {},
crossDomain = false,
global = true
// …more config
}) {
//…do stuff
}
6、遍历map结构

var map = new Map();
map.set(‘first’, ‘hello’);
map.set(‘second’, ‘world’);
for(let [key, value] of map) {
console.log(key + “is” + value)
}
//first is hello , second is world

##字符串的扩展
1、字符的unicode表示法
2、codePointAt（）方法
3、String.fromCodePoint()方法
4、字符串可遍历
5、at（）
6、includes()、startsWith()、endWith()
三个都返回布尔值，第一个检测是否找到参数字符，第二个检测是否在源字符串的头部，第三个检测是否在尾部。
7、repeat()——重复
小数会取整、无穷会报错、NAN等于0、字符串变数字
8、padStart（）、padEnd()
长度补全，接收两个参数，第一个是长度，第二个是补全内容（没有就用空格代替）。
9、模版字符串
字符串用（`）反引号包含，变量用${}包含，JS代码用<%…%>包含，JS表达式用<%=…%>包含
10、标签模版
⑴过滤HTML字符串，防止用户输入恶意内容
⑵多语言转换
11、String.raw()
对附带斜杠的进行转义
